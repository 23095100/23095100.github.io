一、添加自动生成目录

使用 markdownpad2_toc 插件：https://github.com/negrochn/markdownpad2_toc

  效果图如下：

  

  1. 在工具选项卡中，选中选项，进入高级设置，选择 HTMLHead 编辑器进行编辑

  

  1. 复制以下代码进行粘贴
  1<script src="https://cdn.bootcss.com/jquery/3.3.1/jquery.js"></script>
  2<script>
  3    $(document).ready(function() {
  4        // 新增Contains选择器，:contains的变异版，增加忽略大小写功能
  5        $.expr[':'].Contains = function(a, i, m){
  6            return jQuery(a).text().toUpperCase().indexOf(m[3].
  7                toUpperCase()) >= 0;
  8        };
  9
 10        // 创建class为toc的div元素
 11        let $toc = $('<div></div>');
 12        $toc.attr('class', 'toc');
 13
 14        // 获取h1-h6元素
 15        let $headers = $('h1,h2,h3,h4,h5,h6');
 16
 17        // 获取第一个hx元素，注意是dom对象，非jQuery对象
 18        let firstHeader = $headers[0];
 19
 20        if (firstHeader) {
 21            // 获取第一个hx元素的层级
 22            let firstLevel = parseInt(firstHeader.tagName.
 23                replace('H', ''), 10);
 24            // 给a元素唯一的href属性值
 25            let id = `${firstHeader.textContent}-0`;
 26
 27            for (let i = 1; i <= firstLevel; i++) {
 28                // 创建ul和li元素，并添加data-level属性
 29                let $ul = $('<ul></ul>'),
 30                    $li = $('<li></li>');
 31                $li.attr('data-level', i);
 32                $ul.attr('data-level', i).append($li);
 33
 34                // 获取data-level为i-1的li元素
 35                let $pLi = $toc.find(`li[data-level=${i - 1}]`);
 36
 37                if ($pLi.length > 0) {
 38                    // 找到data-level为i-1的元素，直接append到该li元素上即可
 39                    $pLi.append($ul);
 40                } else {
 41                    // 未找到data-level为i-1的元素，说明是顶层的ul，
 42                    // 直接append到class为toc的div元素上
 43                    $toc.append($ul);
 44                }
 45            }
 46
 47            // 找到所属层级的li元素，并添加a元素
 48            $toc.find(`li[data-level=${firstLevel}]`).append(
 49                $(`<a href="#${id}">${firstHeader.textContent}</a>`));
 50            // 与a元素的href对应，用于文档内跳转
 51            $headers.eq(0).attr('id', id);
 52        }
 53
 54        // 从第二个hx元素开始，循环$headers
 55        for (let i = 1; i < $headers.length; i++) {
 56            // 获取当前的hx元素的层级以及上一个hx元素的层级
 57            let curLevel = parseInt($headers[i].tagName.replace('H', ''),
 58                10),
 59                lastLevel = parseInt($headers[i - 1].tagName.
 60                    replace('H', ''), 10);
 61            // 当前hx元素的文本元素
 62            let textContent = $headers[i].textContent,
 63                id = `${textContent}-i`;
 64            // 给a元素唯一的href属性值，与hx元素的id对应
 65            let $a = $(`<a href="#${id}">${textContent}</a>`);
 66
 67            if (lastLevel < curLevel) {
 68                // 如果当前层级比上一个层级大，则从上一个层级+1到当前层级循环，
 69                // 循环添加ul和li元素
 70                for (let j = lastLevel + 1; j <= curLevel; j++) {
 71                    // 创建ul和li元素，并添加data-level属性
 72                    let $u = $('<ul></ul>'),
 73                        $l = $('<li></li>');
 74                    $l.attr('data-level', j);
 75                    $u.attr('data-level', j).append($l);
 76
 77                    // 找到data-level为j-1的最后一个li元素，并添加$u
 78                    $toc.find(`li[data-level=${j - 1}]:last`).append($u);
 79                }
 80                // 找到data-level为curLevel的最后一个li，并添加$a
 81                $toc.find(`li[data-level=${curLevel}]:last`).append($a);
 82            } else if (lastLevel === curLevel) {
 83                // 如果当前层级和上一个层级相等，则找到相同层级的li元素的父元素，
 84                // 创建li元素并添加上去
 85                // 已经存在data-level为curLevel的ul，只需要创建li即可
 86                let $l = $('<li></li>');
 87
 88                // 给li元素添加data-level树形，并添加$a
 89                // 找到data-level为curLevel的最后一个li元素的父元素，添加$l
 90                $toc.find(`li[data-level=${curLevel}]:last`).parent().
 91                    append($l.attr('data-level', curLevel).append($a));
 92            } else {
 93                // 如果当前层级小于上一个层级，则找到最后一个相同层级的li元素的
 94                // 父元素，添加$l
 95                // 找到data-level为curLevel的最后一个li元素,必然存在
 96                let $sublingLi = $toc.
 97                    find(`li[data-level=${curLevel}]:last`);
 98                let $l = $('<li></li>');
 99
100                // 找到$sublingLi的父元素,添加li元素即可
101                $sublingLi.parent().append($l.attr('data-level', curLevel).
102                    append($a));
103            }
104
105            $headers.eq(i).attr('id', id);
106        }
107
108
109        // 清除p元素的文本内容，并将$toc元素append到p元素
110        $('p:Contains([TOC]):first').text('').append($toc);
111    });
112</script>

二、使用 highlightjs 美化代码块

  每次在其他的个人博客上，都能看到很漂亮的代码块，但是自己用 Markdown 语法编写的时候，并没有很好看的那种代码，所以，我这里借用了一下第三方的 js 进行美化一下 Markdown 的代码

highlightjs 官网：https://highlightjs.org/

  效果图如下：

  

  1. 在工具选项卡中，选中选项，进入高级设置，选择 HTMLHead 编辑器进行编辑

  

  1. 添加以下代码块
1<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/highlightjs/cdn-release@10.1.1/build/styles/vs2015.min.css">
2<script src="https://cdn.jsdelivr.net/gh/highlightjs/cdn-release@10.1.1/build/highlight.min.js"></script>
3<script>hljs.initHighlightingOnLoad();</script>
  1. 保存并退出

  

  1. 粘贴代码块进行测试
1<pre><code class="javascript">
2<link rel="stylesheet" href="https://cdn.jsdelivr.net/gh/highlightjs/cdn-release@10.1.1/build/styles/vs2015.min.css">
3<script src="https://cdn.jsdelivr.net/gh/highlightjs/cdn-release@10.1.1/build/highlight.min.js"></script>
4<script>hljs.initHighlightingOnLoad();</script>
5</code></pre>

  highlightjs 会自动检测语言，如果自动检测没有起到作用，那可以在 <code class="javascript"> 中指定语言。需要注意的是

1<pre><code class="javascript">
2</code></pre>

  必须包含在内，class 可以指定语言进行语法高亮显示

  在 MarkdownPad2 中 按 F6 进行预览
